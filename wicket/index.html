LoginPage.java


            public class LoginPage extends WebPage {

                private TextField<String> emailAddress;
                private TextField<String> password;

                public LoginPage(PageParameters pageParameters) {
                    super(pageParameters);

                    add(new FeedbackPanel("feedback"));

                    Form loginForm = new Form<Void>("login-form") {
                        @Override
                        protected void onSubmit() {
                            setResponsePage(HomePage.class);
                        }
                    };
                    loginForm.add(new LoginValidator(emailAddress, password));

                    add(loginForm);


                    emailAddress = new EmailTextField("username", new Model<String>());
                    emailAddress.add(EmailAddressValidator.getInstance());
                    emailAddress.setRequired(true);
                    loginForm.add(emailAddress);


                    password = new PasswordTextField("password", new Model<String>());
                    loginForm.add(password);
                }
            }

BaseTemplatePage.java

                
                public abstract class BaseTemplatePage extends WebPage {

                    public BaseTemplatePage(PageParameters parameters) {
                        super(parameters);

                        Label lblPageTitle = new Label("pageTitle", getPageTitle());

                        SearchPanel searchPanel = new SearchPanel("search-panel");
                        add(searchPanel);

                        NavMenuPanel navMenuPanel = new NavMenuPanel("nav-menu-panel");
                        add(navMenuPanel);
                    }

                    public abstract String getPageTitle();
                }

Wicket Tester - A helper class to ease unit testing of Wicket applications without the need for a servlet container.


            @RunWith(SpringJUnit4ClassRunner.class)
            @ContextConfiguration(locations = { "/applicationContext-test.xml" })
            public class TestHomePage {
                private WicketTester tester;

                @Before
                public void beforeTest() {
                    final ApplicationContextMock mockContext = new ApplicationContextMock();

                    addSpringBeans(mockContext);
                    UmsApplication webApplication = new UmsApplication() {
                        @Override
                        protected SpringComponentInjector createSpringComponentInjector() {
                            return  new SpringComponentInjector(this, mockContext, true);
                        }
                    };

                    tester = new WicketTester(webApplication);
                }


                @Test
                public void testSendMMS() {
                    tester.startPage(HomePage.class);

                    FormTester formTester = tester.newFormTester(CONTENT_PANEL + ":form");
                    Form form = formTester.getForm();

                    TextArea txtMessageContent = (TextArea) form.get("textMessage");
                    formTester.setValue(txtMessageContent, "Wicket is nice I tell you.");

                    TextField txtFrom = (TextField) form.get("from");
                    formTester.setValue(txtFrom, "+40722533281");

                    AjaxButton btnSendMessage = (AjaxButton) form.get("submit");
                    tester.executeAjaxEvent(btnSendMessage, "onclick");

                    tester.assertNoErrorMessage();

                    tester.assertComponent(CONTENT_PANEL, ConfirmationPanel.class);
                }
            }
