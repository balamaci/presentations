                public class LoginPage extends WebPage {

                private TextField<String> emailAddress;
                private TextField<String> password;

                public LoginPage(PageParameters pageParameters) {
                super(pageParameters);

                add(new FeedbackPanel("feedback"));

                Form loginForm = new Form<Void>("login-form") {
                @Override
                protected void onSubmit() {
                if ( ! continueToOriginalDestination()) {
                setResponsePage(HomePage.class);
                }
                }
                };
                loginForm.add(new LoginValidator(emailAddress, password));

                add(loginForm);


                emailAddress = new EmailTextField("email", new Model<String>());
                emailAddress.add(EmailAddressValidator.getInstance());
                emailAddress.setRequired(true);
                loginForm.add(emailAddress);


                password = new PasswordTextField("password", new Model<String>());
                loginForm.add(password);

                }
                }
